                      border: pw.Border.all(width: 0.8, color: PdfColors.green700),
                    ),
                    alignment: pw.Alignment.center,
                    child: pw.Text(
                      '✓',
                      style: pw.TextStyle(
                        fontSize: 7,
                        fontWeight: pw.FontWeight.bold,
                        color: PdfColors.white,
                      ),
                    ),
                  );
                } else if (isPending) {
                  // Szary pusty kwadrat
                  statusWidget = pw.Container(
                    width: 10,
                    height: 10,
                    decoration: pw.BoxDecoration(
                      border: pw.Border.all(width: 1.2, color: PdfColors.grey600),
                    ),
                  );
                } else {
                  // Tekst "n.d." 
                  statusWidget = pw.Text(
                    'n.d.',
                    style: pw.TextStyle(
                      fontSize: 6,
                      color: PdfColors.grey500,
                      fontStyle: pw.FontStyle.italic,
                    ),
                  );
                }
                
                return pw.TableRow(
                  decoration: pw.BoxDecoration(color: rowColor),
                  children: [
                    pw.Padding(
                      padding: const pw.EdgeInsets.symmetric(horizontal: 4, vertical: 2),
                      child: pw.Text(entry.key, style: pw.TextStyle(fontSize: 8)),
                    ),
                    pw.Container(
                      padding: const pw.EdgeInsets.all(2),
                      child: statusWidget,
                    ),
                  ],
                );
              }).toList(),
            ],
          ),
          pw.SizedBox(height: 10),
          
          // POLE USTERKI
          pw.Text(
            'Usterki:',
            style: pw.TextStyle(fontSize: 9, fontWeight: pw.FontWeight.bold),
          ),
          pw.SizedBox(height: 2),
          pw.Container(
            height: 200,
            decoration: pw.BoxDecoration(
              border: pw.Border.all(width: 1, color: PdfColors.black),
            ),
          ),
          pw.SizedBox(height: 10),
          
          // POLE UWAGI
          pw.Text(
            'Uwagi:',
            style: pw.TextStyle(fontSize: 9, fontWeight: pw.FontWeight.bold),
          ),
          pw.SizedBox(height: 2),
          pw.Container(
            height: 200,
            decoration: pw.BoxDecoration(
              border: pw.Border.all(width: 1, color: PdfColors.black),
            ),
          ),
        ];
      },
    ),
  );

  await Printing.layoutPdf(onLayout: (_) async => doc.save());
}
